function bingo(ticket, win){
  let counter = 0;
  for(let value of ticket) {
    value[0].split('').forEach(item => {
      if(item.charCodeAt() === value[1]) {
        return counter++
      };
    });
  }
  if(counter >= win) {
    return ('Winner!');
  } else {
    return ('Loser!')
  }
}
